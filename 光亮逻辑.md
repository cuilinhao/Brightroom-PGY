```80:97:Sources/BrightroomUI/builtin/ClassicImageEdit/Components/FilterControl/ClassicImageEditExposureControl.swift
@objc
private func valueChanged() {

  let value = slider.transition(in: FilterExposure.range)
  guard value != 0 else {
    viewModel.editingStack.set(filters: {
      $0.exposure = nil          // 值为0时彻底清空，不留空滤镜
    })
    return
  }    
      
  viewModel.editingStack.set(filters: {
    var f = FilterExposure()
    f.value = value
    $0.exposure = f             // 赋值新的曝光滤镜
  })
  
}
```

### 第三层：数据模型层

**FilterExposure** — 滤镜参数结构体：

文件：`FilterExposure.swift`

```25:46:Sources/BrightroomEngine/Filter/FilterExposure.swift
public struct FilterExposure : Filtering, Equatable, Codable, Sendable {
  
  public static let range: ParameterRange<Double, FilterExposure> = .init(min: -1.8, max: 1.8)
  
  public var value: Double = 0
  
  public init() {
    
  }
  
  public func apply(to image: CIImage, sourceImage: CIImage) -> CIImage {
    return
      image
        .applyingFilter(
          "CIExposureAdjust",
          parameters: [
            kCIInputEVKey: value as AnyObject   // 直接对接 Core Image 曝光滤镜
          ]
    )
  }
  
}
```

**Edit.Filters** — 统一管理所有滤镜，`exposure` 是可选属性，nil 代表未调整：

```49:108:Sources/BrightroomEngine/Core/EditingStack.Edit.swift
public struct Filters: Equatable {

  public var preset: FilterPreset?
  // ...
  public var exposure: FilterExposure?
  // ...

  func makeFilters() -> [AnyFilter] {
    return (
      ([
        preset?.asAny(),
        exposure?.asAny(),     // exposure 在滤镜链中排第二，preset 之后
        brightness?.asAny(),
        temperature?.asAny(),
        // ...更多滤镜
      ] as [AnyFilter?])
      + additionalFilters
    )
    .compactMap { $0 }         // nil 的自动过滤掉
  }
  
  public func apply(to ciImage: CIImage) -> CIImage {
    makeFilters().reduce(ciImage) { (image, filter) -> CIImage in
      filter.apply(to: image, sourceImage: image)   // 链式 reduce 依次应用
    }
  }
}
```

### 第四层：渲染层

**预览渲染**（拖动时实时看到效果）：

- `EditingStack.Loaded.currentEdit` 的 `didSet` → `editingPreviewImage = currentEdit.filters.apply(editingSourceImage)` → 立刻重算整条 Core Image 滤镜链
- `ImagePreviewView` 监听 `editingPreviewImage` 变化 → `MetalImageView.postProcessing = filters.apply` → 用 `CIContext(mtlDevice:)` 渲染到 Metal 纹理上屏

**导出渲染**（`makeRenderer().render()`）：

- `BrightRoomImageRenderer.edit.modifiers = edit.makeFilters()`
- `modifiers.reduce(sourceCIImage)` → 每个滤镜依次 `apply` → 含 `FilterExposure` → `CIExposureAdjust` → 最终 `createCGImage`

### 端到端调用链总结

```
用户点击 exposureButton
  └─ brightness() → push(ClassicImageEditExposureControl)

用户拖动 slider
  └─ valueChanged()
       └─ slider.transition(in: FilterExposure.range)  // UISlider步进值 → EV值(-1.8~1.8)
       └─ editingStack.set(filters: { $0.exposure = FilterExposure(value: x) })

EditingStack
  └─ applyIfChanged → loadedState.currentEdit.filters.exposure 更新
  └─ currentEdit.didSet
       └─ editingPreviewImage = filters.apply(editingSourceImage)
            └─ makeFilters().reduce  →  FilterExposure.apply
                 └─ CIImage.applyingFilter("CIExposureAdjust", kCIInputEVKey: value)

MetalImageView
  └─ postProcessing(resolvedImage) → CIContext(mtlDevice:).render → 上屏

用户点 Done
  └─ editingStack.takeSnapshot()  // 确认提交当前编辑状态

用户点 Cancel
  └─ editingStack.revertEdit()    // 回滚到进入时的快照
```

**几个设计要点值得关注：**

1. `exposure == nil` 表示「未调整」，而不是 `value = 0`。这样 `makeFilters()` 里 `compactMap` 直接跳过，不引入多余的 Core Image 节点。

2. 滤镜链顺序是硬编码在 `makeFilters()` 里的，`exposure` 固定在 `preset` 之后、`brightness` 之前，顺序影响最终效果。

3. 预览走 Metal (`CIContext` on MTLDevice)，导出走软件渲染 (`useSoftwareRenderer: true`)，两条路最终调用的是同一个 `FilterExposure.apply` 方法，保证预览与导出一致。
